# Function definition
def greet(name: str) -> str:
  return f"Hello, {name}!"

# List comprehension
squares = [x**2 for x in range(10)]

# Dictionary
band_info = {
  "name": "The Beatles",

}

# Class definition
class Musician:
  def __init__(self, name: str, instrument: str):
    self.name = name
    self.instrument = instrument

  def play(self) -> str:
    return f"{self.name} is playing the {self.instrument}"

# Using the class
john = Musician("John Lennon", "guitar")
print(john.play())

# Lambda function
add = lambda x, y: x + y

# Decorator
def uppercase_decorator(func):
  def wrapper(*args, **kwargs):
    result = func(*args, **kwargs)
    return result.upper()
  return wrapper

@uppercase_decorator
def say_hi(name: str) -> str:
  return f"Hi, {name}"

print(say_hi("Paul"))

# Exception handling
try:
  result = 10 / 0
except ZeroDivisionError as e:
  print(f"Error: {e}")

# Generator
def countdown(n: int):
  while n > 0:
    yield n
    n -= 1

for number in countdown(5):
  print(number)
